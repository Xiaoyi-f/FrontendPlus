对象基础与属性操作

对象是JavaScript中最核心的概念之一，是存储键值对的数据结构。对象可以包含属性和方法，是JavaScript编程的基础。

对象概览

  • 对象创建：多种创建方式（字面量、构造函数等）
  • 属性访问：点表示法和括号表示法
  • 属性操作：添加、修改、删除属性
  • 对象方法：对象中的函数属性
  • 对象遍历：for...in循环和Object方法
  • 对象解构：ES6解构赋值

创建对象

JavaScript提供了多种创建对象的方式，每种方式都有其适用场景。

创建方式 语法 特点 适用场景
对象字面量 {key: value} 最常用，简洁 通用创建方式
构造函数 new Object() 传统方式 动态创建
Object.create() Object.create(proto) 指定原型创建 继承场景
类语法 class MyClass {} ES6+语法糖 面向对象编程
// 对象字面量（推荐方式）
const person = {
  name: "张三",
  age: 30,
  city: "北京",
  isMarried: false,
  hobbies: ["阅读", "游泳", "编程"]
};

// 使用构造函数创建对象
const car = new Object();
car.brand = "丰田";
car.model = "卡罗拉";
car.year = 2022;

// 使用Object.create()创建对象
const animal = Object.create(null);
animal.name = "老虎";
animal.type = "哺乳动物";

// 从已有对象创建新对象
const student = Object.create(person);
student.grade = "高一";
student.school = "一中";

console.log(person); // {name: "张三", age: 30, city: "北京", isMarried: false, hobbies: Array(3)}
console.log(car); // {brand: "丰田", model: "卡罗拉", year: 2022}
console.log(animal); // {name: "老虎", type: "哺乳动物"}

属性访问和修改

对象属性可以通过两种方式访问：点表示法和括号表示法。

const user = {
  firstName: "李",
  lastName: "四",
  age: 25,
  "full-name": "李四", // 包含特殊字符的键名
  preferences: {
    theme: "dark",
    language: "zh-CN"
  }
};

// 点表示法访问属性
console.log(user.firstName); // "李"
console.log(user.age); // 25

// 括号表示法访问属性
console.log(user['lastName']); // "四"
console.log(user["full-name"]); // "李四" - 只能用括号表示法访问

// 访问嵌套属性
console.log(user.preferences.theme); // "dark"
console.log(user.preferences['language']); // "zh-CN"

// 修改属性值
user.age = 26;
user['firstName'] = "王";

// 添加新属性
user.email = "wang@example.com";
user['phone'] = "13800138000";

// 使用变量作为属性名（ES6计算属性名）
const dynamicKey = 'location';
user[dynamicKey] = "上海";

console.log(user); // 更新后的对象
点表示法 vs 括号表示法
const obj = {
  "first-name": "张",
  age: 30,
  city: "北京"
};

// 点表示法 - 适用于标准标识符
console.log(obj.age); // 30
console.log(obj.city); // "北京"

// 括号表示法 - 适用于所有情况
console.log(obj["first-name"]); // "张" - 包含特殊字符的键名
console.log(obj['age']); // 30

// 使用变量作为键名
const key = 'city';
console.log(obj[key]); // "北京"
属性访问方式对比
  • 点表示法：简洁，适用于标准标识符
  • 括号表示法：灵活，适用于所有键名，支持动态键名
  • 安全访问：使用可选链操作符?.避免错误
  • 默认值：结合空值合并操作符??设置默认值

对象方法

对象中的函数属性称为方法，方法可以访问对象的属性并执行操作。

const calculator = {
  value: 0,

  // ES5方法定义
  add: function(num) {
    this.value += num;
    return this; // 返回this以支持链式调用
  },

  // ES6简写方法定义
  subtract(num) {
    this.value -= num;
    return this;
  },

  multiply(num) {
    this.value *= num;
    return this;
  },

  getValue() {
    return this.value;
  },

  reset() {
    this.value = 0;
    return this;
  }
};

// 使用计算器
calculator.add(10).multiply(2).subtract(5);
console.log(calculator.getValue()); // 15
calculator.reset();
console.log(calculator.getValue()); // 0
对象属性操作
const book = {
  title: "JavaScript高级程序设计",
  author: "Nicholas C. Zakas",
  year: 2020
};

// 检查属性是否存在
console.log('title' in book); // true
console.log(book.hasOwnProperty('title')); // true

// 获取对象的所有属性名
const keys = Object.keys(book);
console.log(keys); // ["title", "author", "year"]

// 获取对象的所有属性值
const values = Object.values(book);
console.log(values); // ["JavaScript高级程序设计", "Nicholas C. Zakas", 2020]

// 获取属性名和值的键值对
const entries = Object.entries(book);
console.log(entries); // [["title", "JavaScript高级程序设计"], ["author", "Nicholas C. Zakas"], ["year", 2020]]

// 删除属性
delete book.year;
console.log(book); // {title: "JavaScript高级程序设计", author: "Nicholas C. Zakas"}

// 防止对象被修改
Object.freeze(book); // 冻结对象，无法添加、删除或修改属性
book.pages = 800; // 无效操作
console.log(book.pages); // undefined

// 防止添加新属性（但仍可修改现有属性）
const config = {
  debug: true,
  timeout: 5000
};
Object.preventExtensions(config);
config.newProp = 'value'; // 无效操作（在严格模式下会报错）

对象遍历

有多种方法可以遍历对象的属性。

const laptop = {
  brand: "Apple",
  model: "MacBook Pro",
  year: 2021,
  specifications: {
    cpu: "M1",
    ram: "16GB",
    storage: "512GB SSD"
  }
};

// for...in循环遍历对象的可枚举属性
for (let key in laptop) {
  if (laptop.hasOwnProperty(key)) {
    console.log(`${key}: ${laptop[key]}`);
  }
}

// 使用Object.keys()遍历
Object.keys(laptop).forEach(key => {
  console.log(`Key: ${key}, Value: ${laptop[key]}`);
});

// 使用Object.entries()遍历
Object.entries(laptop).forEach(([key, value]) => {
  console.log(`${key} = ${value}`);
});

// 深度遍历嵌套对象
function deepLog(obj, prefix = '') {
  for (let key in obj) {
    if (typeof obj[key] === 'object' && obj[key] !== null && !Array.isArray(obj[key])) {
      deepLog(obj[key], `${prefix}${key}.`);
    } else {
      console.log(`${prefix}${key}: ${obj[key]}`);
    }
  }
}
deepLog(laptop);

ES6+对象特性

ES6及后续版本引入了许多对象相关的特性，使对象操作更加便捷。

const name = "张三";
const age = 25;
const city = "上海";

// 属性简写
const person = {
  name, // 等价于 name: name
  age, // 等价于 age: age
  city // 等价于 city: city
};

// 方法简写
const user = {
  name: "李四",
  getName() {
    return this.name;
  },
  setName(newName) {
    this.name = newName;
  }
};

// 计算属性名
const keyPrefix = 'user';
const dynamicObj = {
  [`${keyPrefix}_name`]: "王五",
  [`${keyPrefix}_age`]: 30,
  ['computed_' + new Date().getDay()]: 'Today'
};
console.log(dynamicObj); // {user_name: "王五", user_age: 30, computed_1: "Today"}

// 对象展开运算符
const defaults = {
  theme: 'light',
  language: 'en',
  notifications: true
};

const userSettings = {
  language: 'zh-CN',
  autoSave: false
};

// 合并对象
const finalSettings = { ...defaults, ...userSettings };
console.log(finalSettings); // {theme: "light", language: "zh-CN", notifications: true, autoSave: false}

// 对象解构
const product = {
  id: 1,
  name: "笔记本电脑",
  price: 5999,
  category: "电子产品"
};

const { id, name, price } = product;
console.log(id, name, price); // 1 "笔记本电脑" 5999

// 解构时重命名
const { name: productName, price: productPrice } = product;
console.log(productName, productPrice); // "笔记本电脑" 5999

// 解构时设置默认值
const { name: n, warranty = '1年' } = product;
console.log(n, warranty); // "笔记本电脑" "1年"

// 嵌套解构
const company = {
  name: "科技有限公司",
  address: {
    city: "北京",
    district: "海淀区",
    street: "中关村大街"
  },
  employees: 50
};

const { address: { city, district } } = company;
console.log(city, district); // "北京" "海淀区"
