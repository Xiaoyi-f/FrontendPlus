字符串处理

字符串是JavaScript中最重要的数据类型之一，用于表示文本数据。JavaScript提供了丰富的字符串方法和操作方式，让我们能够轻松地创建、操作和处理字符串。掌握字符串处理是JavaScript编程的基础技能。

字符串概览

  • 字符串创建：单引号、双引号、反引号
  • 字符串访问：索引、charAt()、at()
  • 字符串操作：拼接、截取、替换、分割等
  • 字符串搜索：indexOf()、includes()、startsWith()等
  • 字符串格式化：模板字符串、padStart()等
  • ES2022/2023新方法：at()、replaceAll()等

创建字符串

JavaScript提供多种方式创建字符串：

// 1. 单引号
const str1 = 'Hello, World!';

// 2. 双引号
const str2 = "JavaScript is awesome!";

// 3. 反引号（模板字符串，ES6）
const name = "张三";
const str3 = `Hello, ${name}! Today is ${new Date().toLocaleDateString()}`;

// 4. String构造函数
const str4 = new String("构造函数创建的字符串"); // 返回String对象
const str5 = String(123); // 将其他类型转换为字符串

// 字符串是不可变的
let text = "Hello";
text[0] = "h"; // 尝试修改字符串中的字符
console.log(text); // "Hello" - 原字符串没有改变

// 需要创建新字符串来实现修改
text = "h" + text.slice(1); // "hello"
console.log(text); // "hello"

字符串访问和属性

可以使用多种方式访问字符串中的字符和获取字符串信息：

const text = "Hello, JavaScript!";

// 获取字符串长度
console.log(text.length); // 18

// 访问字符 - 方括号表示法（ES5.1）
console.log(text[0]); // "H"
console.log(text[7]); // "J"

// 使用charAt()方法
console.log(text.charAt(0)); // "H"
console.log(text.charAt(100)); // "" (空字符串)

// ES2022的at()方法 - 支持负索引
console.log(text.at(-1)); // "!" (最后一个字符)
console.log(text.at(-2)); // "t" (倒数第二个字符)

// 遍历字符串
for (const char of text) {
  console.log(char);
}

// 使用Array.from()将字符串转换为字符数组
const charArray = Array.from(text);
console.log(charArray); // ['H', 'e', 'l', 'l', 'o', ',', ' ', 'J', ...]

// 使用扩展运算符将字符串转换为字符数组
const charArray2 = [...text];
console.log(charArray2); // ['H', 'e', 'l', 'l', 'o', ',', ' ', 'J', ...]

字符串基本操作

字符串的拼接、截取、转换等基本操作：

const str = "Hello, JavaScript World!";

// 字符串拼接
const greeting = "Hello" + " " + "World";
const templateGreeting = `Hello ${"World"}`;
const concatGreeting = "Hello".concat(" ", "World");

// 截取字符串
console.log(str.slice(0, 5)); // "Hello" (从索引0到5，不包括5)
console.log(str.substring(0, 5)); // "Hello" (与slice类似，但参数可以交换)
console.log(str.substr(0, 5)); // "Hello" (从索引0开始，取5个字符，已废弃)

// 使用slice()的负索引
console.log(str.slice(-5)); // "World" (从倒数第5个字符到末尾)
console.log(str.slice(-5, -1)); // "Worl" (从倒数第5个到倒数第1个，不包括)

// 大小写转换
console.log(str.toUpperCase()); // "HELLO, JAVASCRIPT WORLD!"
console.log(str.toLowerCase()); // "hello, javascript world!"

// 去除空白字符
const spacedStr = " Hello, World! ";
console.log(spacedStr.trim()); // "Hello, World!"
console.log(spacedStr.trimStart()); // "Hello, World! "
console.log(spacedStr.trimEnd()); // " Hello, World!"

// 替换字符串
console.log(str.replace("JavaScript", "TypeScript")); // "Hello, TypeScript World!"
console.log(str.replace(/o/g, "0")); // "Hell0, Javascr ipt W0rld!" (全局替换)

// ES2021的replaceAll()方法
console.log(str.replaceAll('o', '0')); // "Hell0, Javascr ipt W0rld!"

字符串搜索方法

JavaScript提供了多种方法来搜索和检测字符串中的内容：

const text = "JavaScript is a programming language";

// indexOf() - 查找子字符串第一次出现的位置
console.log(text.indexOf("a")); // 1 (从0开始的索引)
console.log(text.indexOf("a", 2)); // 3 (从索引2开始查找)
console.log(text.indexOf("Python")); // -1 (未找到)

// lastIndexOf() - 查找子字符串最后一次出现的位置
console.log(text.lastIndexOf("a")); // 30

// includes() - 检查字符串是否包含子字符串
console.log(text.includes("JavaScript")); // true
console.log(text.includes("Python")); // false
console.log(text.includes("a", 10)); // true (从索引10开始查找)

// startsWith() - 检查字符串是否以特定子字符串开始
console.log(text.startsWith("JavaScript")); // true
console.log(text.startsWith("a", 1)); // true (从索引1开始检查)

// endsWith() - 检查字符串是否以特定子字符串结束
console.log(text.endsWith("language")); // true
console.log(text.endsWith("programming", 20)); // true (检查前20个字符是否以"programming"结束)

// search() - 使用正则表达式搜索
console.log(text.search(/JAVA/i)); // 0 (找到匹配项的索引)

// match() - 使用正则表达式匹配
const matches = text.match(/a/g); // 全局匹配所有'a'
console.log(matches); // ['a', 'a', 'a', 'a', 'a']
方法 描述 返回值 示例
charAt(index) 返回指定索引处的字符 字符 str.charAt(0)
indexOf(searchValue) 查找子字符串第一次出现的位置 索引或-1 str.indexOf('a')
substring(start, end) 提取子字符串 新字符串 str.substring(0, 5)
slice(start, end) 提取子字符串（支持负索引） 新字符串 str.slice(-5)
toUpperCase() 转换为大写 新字符串 str.toUpperCase()
toLowerCase() 转换为小写 新字符串 str.toLowerCase()
trim() 去除首尾空白字符 新字符串 str.trim()
split(separator) 按分隔符分割为数组 数组 str.split(' ')
replace(search, replacement) 替换子字符串 新字符串 str.replace('old', 'new')
concat(str1, str2, ...) 连接字符串 新字符串 str.concat(' more', ' text')

字符串分割与连接

将字符串分割成数组或将数组连接成字符串：

const sentence = "JavaScript is a powerful programming language";

// split() - 按分隔符分割字符串
const words = sentence.split(' ');
console.log(words); // ["JavaScript", "is", "a", "powerful", "programming", "language"]

// 按多个分隔符分割
const complexStr = "apple,banana;orange:grape";
const fruits = complexStr.split(/[,;:]/);
console.log(fruits); // ["apple", "banana", "orange", "grape"]

// 按字符分割
const chars = sentence.split('');
console.log(chars.length); // 42 (字符数量)

// 限制分割数量
const limitedWords = sentence.split(' ', 3);
console.log(limitedWords); // ["JavaScript", "is", "a"]

// join() - 将数组元素连接为字符串
const fruitsArray = ["apple", "banana", "orange"];
const fruitsString = fruitsArray.join(', ');
console.log(fruitsString); // "apple, banana, orange"

// 无分隔符连接
const joinedWithoutSeparator = fruitsArray.join('');
console.log(joinedWithoutSeparator); // "applebananaorange"
slice() vs substring() vs substr()
  • slice(start, end)：支持负索引，end参数不包含在结果中
  • substring(start, end)：不支持负索引，参数会自动交换（如果start > end）
  • substr(start, length)：第二个参数是长度，已废弃，推荐使用slice()
  • 推荐使用slice()，因为它功能最全面且行为最可预测
字符串方法的不可变性
  • 所有字符串方法都不会修改原字符串
  • 所有操作都会返回新字符串
  • 这符合函数式编程原则，避免副作用
  • 如果需要修改原字符串，需要将结果重新赋值给变量

字符串格式化和填充

格式化字符串以满足特定的显示要求：

// padStart() 和 padEnd() - 字符串填充
console.log('5'.padStart(3, '0')); // "005"
console.log('5'.padEnd(3, '0')); // "500"
console.log('Hello'.padStart(10, '*')); // "*****Hello"

// repeat() - 重复字符串
console.log('abc'.repeat(3)); // "abcabcabc"

// 模板字符串（ES6）
const name = "Alice";
const age = 30;
const profile = `Name: ${name}, Age: ${age}`;
console.log(profile); // "Name: Alice, Age: 30"

// 模板字符串中的表达式
const calculation = `2 + 2 = ${2 + 2}`;
console.log(calculation); // "2 + 2 = 4"

// 多行字符串（模板字符串）
const multiline = `
这是第一行
这是第二行
这是第三行
`;
console.log(multiline);

// 标签模板（高级用法）
function highlight(strings, ...values) {
  let result = '';
  strings.forEach((string, i) => {
    result += string;
    if (i < values.length) {
      result += `${values[i]}`;
    }
  });
  return result;
}

const highlighted = highlight`Hello, ${name}! You are ${age} years old.`;
console.log(highlighted); // "Hello, Alice! You are 30 years old."

URL处理示例
// URL处理函数
function parseURL(url) {
  // 移除末尾的斜杠
  if (url.endsWith('/')) {
    url = url.slice(0, -1);
  }

  const parts = url.split('/');
  const protocol = parts[0].includes(':') ? parts[0].replace(':', '') : null;
  const host = protocol ? parts[2] : parts[0] || '';
  const path = protocol ? parts.slice(3).join('/') : parts.slice(1).join('/') || '';

  return { protocol, host, path };
}

const urlInfo = parseURL('https://example.com/users/profile');
console.log(urlInfo); // {protocol: "https", host: "example.com", path: "users/profile"}

// 构建查询字符串
function buildQueryString(params) {
  return Object.entries(params)
    .map(([key, value]) => `${encodeURIComponent(key)}=${encodeURIComponent(value)}`)
    .join('&');
}

const query = buildQueryString({ name: 'Alice', age: 30, city: 'New York' });
console.log(query); // "name=Alice&age=30&city=New%20York"
文本处理示例
// 文本处理工具函数
function processText(text) {
  // 清理文本 - 去除多余空格并标准化换行
  const cleaned = text
    .replace(/\s+/g, ' ') // 多个空白字符替换为单个空格
    .trim(); // 去除首尾空格

  // 统计单词
  const words = cleaned.split(/\s+/).filter(word => word.length > 0);
  const wordCount = words.length;

  // 统计字符（不包括空格）
  const charCount = cleaned.replace(/\s/g, '').length;

  // 找出最长的单词
  const longestWord = words.reduce((longest, current) =>
    current.length > longest.length ? current : longest, '');

  return {
    original: text,
    cleaned: cleaned,
    wordCount: wordCount,
    charCount: charCount,
    longestWord: longestWord,
    longestWordLength: longestWord.length
  };
}

const textStats = processText(`
JavaScript is a versatile programming language
that is widely used for web development.
`);

console.log(textStats);
