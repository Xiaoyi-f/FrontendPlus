函数定义与调用

函数是JavaScript中最重要的概念之一，它是可重用的代码块，可以接收输入参数并返回结果。函数使代码更加模块化、可维护和可重用。

函数概览

  • 函数声明：使用function关键字声明
  • 函数表达式：将函数赋值给变量
  • 箭头函数：ES6引入的简洁语法
  • 参数处理：默认参数、剩余参数、参数解构
  • 作用域：函数作用域、块作用域、闭包
  • 调用方式：普通调用、方法调用、构造函数调用

函数定义方式

JavaScript提供了多种定义函数的方式，每种方式都有其特定的用途和特点。

定义方式 语法 特点 适用场景
函数声明 function name() {} 函数提升，可立即调用 通用，主要定义方式
函数表达式 const name = function() {} 不提升，需先定义后调用 作为参数传递，匿名函数
箭头函数 const name = () => {} 不绑定this，简洁语法 回调函数，保持this上下文
构造函数 new Function('a', 'b', 'return a+b') 动态创建，性能较低 动态函数创建

函数声明

函数声明是定义函数的最常见方式，使用function关键字。

// 函数声明
function greet(name) {
  return `你好, ${name}!`;
}

// 函数声明会被提升到作用域顶部，所以可以在声明前调用
console.log(greet("张三")); // "你好, 张三!"

// 带默认参数的函数声明
function multiply(a, b = 1) {
  return a * b;
}
console.log(multiply(5, 3)); // 15
console.log(multiply(5)); // 5 (使用默认参数)

// 函数声明可以包含多个参数和复杂逻辑
function calculateArea(shape, ...dimensions) {
  switch (shape) {
    case "rectangle":
      return dimensions[0] * dimensions[1]; // 长×宽
    case "circle":
      return Math.PI * dimensions[0] * dimensions[0]; // πr²
    default:
      return 0;
  }
}
console.log(calculateArea("rectangle", 5, 3)); // 15
console.log(calculateArea("circle", 4)); // 约50.27

函数表达式

函数表达式将函数赋值给一个变量，也可以是匿名函数。

// 函数表达式
const add = function(a, b) {
  return a + b;
};

// 函数表达式不会被提升，不能在定义前调用
// console.log(add(2, 3)); // TypeError: Cannot access 'add' before initialization
console.log(add(2, 3)); // 5

// 立即执行函数表达式 (IIFE)
(function() {
  console.log("这个函数立即执行");
})();

// 使用IIFE创建私有作用域
const counter = (function() {
  let count = 0;

  return {
    increment: function() {
      count++;
      return count;
    },
    decrement: function() {
      count--;
      return count;
    },
    getCount: function() {
      return count;
    }
  };
})();

console.log(counter.increment()); // 1
console.log(counter.increment()); // 2
console.log(counter.decrement()); // 1
console.log(counter.getCount()); // 1
函数声明 vs 函数表达式
// 函数声明 - 会被提升
hoistedFunc(); // "我是被提升的函数"

function hoistedFunc() {
  console.log("我是被提升的函数");
}

// 函数表达式 - 不会被提升
// notHoistedFunc(); // TypeError: Cannot access 'notHoistedFunc' before initialization

const notHoistedFunc = function() {
  console.log("我不是被提升的函数");
};
函数的命名
  • 函数声明：必须有名称
  • 函数表达式：可以是匿名函数或命名函数
  • 命名函数表达式：有助于调试和递归
  • 最佳实践：使用有意义的函数名

箭头函数

箭头函数是ES6引入的简洁语法，有独特的特性和限制。

// 基本箭头函数
const square = (x) => {
  return x * x;
};

// 单参数可省略括号
const double = x => x * 2;

// 单表达式可省略return和花括号
const add = (a, b) => a + b;

// 返回对象字面量需要括号
const createUser = (name, age) => ({ name, age });

console.log(square(4)); // 16
console.log(double(5)); // 10
console.log(add(3, 7)); // 10
console.log(createUser("李四", 25)); // {name: "李四", age: 25}

// 箭头函数与this的关系
function Timer() {
  this.seconds = 0;

  setInterval(() => {
    this.seconds++; // 箭头函数保持this指向Timer实例
    console.log(this.seconds);
  }, 1000);
}

// 普通函数会改变this指向
function TimerWithRegularFunction() {
  this.seconds = 0;

  setInterval(function() {
    // this不再指向TimerWithRegularFunction实例
    console.log("this.seconds在普通函数中是:", this.seconds);
  }, 1000);
}
参数处理技巧
// 默认参数
function greetUser(name = "访客", greeting = "你好") {
  return `${greeting}, ${name}!`;
}
console.log(greetUser()); // "你好, 访客!"
console.log(greetUser("王五")); // "你好, 王五!"
console.log(greetUser("赵六", "欢迎")); // "欢迎, 赵六!"

// 剩余参数
function sum(...numbers) {
  return numbers.reduce((total, num) => total + num, 0);
}
console.log(sum(1, 2, 3, 4, 5)); // 15
console.log(sum(10, 20)); // 30

// 参数解构
function displayUser({ name, age, email = "未提供邮箱" }) {
  console.log(`姓名: ${name}, 年龄: ${age}, 邮箱: ${email}`);
}
const user = { name: "孙七", age: 30 };
displayUser(user); // "姓名: 孙七, 年龄: 30, 邮箱: 未提供邮箱"

// 解构参数与默认值结合
function createUser({ name = "匿名", role = "user", permissions = [] } = {}) {
  return { name, role, permissions };
}
console.log(createUser()); // {name: "匿名", role: "user", permissions: []}
console.log(createUser({ name: "周八" })); // {name: "周八", role: "user", permissions: []}

函数调用方式

函数可以以多种方式调用，每种方式会影响this的绑定。

const person = {
  name: "张三",
  age: 25,
  greet: function() {
    console.log(`我是${this.name}, 今年${this.age}岁`);
  }
};

// 方法调用 - this指向对象
person.greet(); // "我是张三, 今年25岁"

// 普通函数调用 - this在严格模式下是undefined，非严格模式下是全局对象
const greetFunc = person.greet;
// greetFunc(); // 严格模式下this是undefined，会导致错误

// 使用call、apply、bind调用
const otherPerson = { name: "李四", age: 30 };
person.greet.call(otherPerson); // "我是李四, 今年30岁"
person.greet.apply(otherPerson); // "我是李四, 今年30岁"

const boundGreet = person.greet.bind(otherPerson);
boundGreet(); // "我是李四, 今年30岁"

高阶函数

高阶函数是接受其他函数作为参数或返回函数的函数，是函数式编程的基础。

// 高阶函数示例
function createMultiplier(multiplier) {
  return function(number) {
    return number * multiplier;
  };
}

const double = createMultiplier(2);
const triple = createMultiplier(3);

console.log(double(5)); // 10
console.log(triple(4)); // 12

// 数组方法作为高阶函数
const numbers = [1, 2, 3, 4, 5];

// map: 对每个元素应用函数
const doubled = numbers.map(num => num * 2);
console.log(doubled); // [2, 4, 6, 8, 10]

// filter: 过滤元素
const evens = numbers.filter(num => num % 2 === 0);
console.log(evens); // [2, 4]

// reduce: 将数组归约为单个值
const sum = numbers.reduce((acc, num) => acc + num, 0);
console.log(sum); // 15

// 自定义高阶函数
function executeTwice(func, arg) {
  func(arg);
  func(arg);
}

executeTwice(console.log, "Hello"); // 输出"Hello"两次
