Grid布局核心语法速查表

/* 容器核心属性 */
grid-template-columns/rows: 尺寸/函数;  // 显式行列
grid-template-areas: "区域命名";        // 区域定义
grid-auto-rows/columns: 尺寸;           // 隐式行列
grid-auto-flow: row | column | dense;   // 自动流方向
justify/align-content: 对齐方式;        // 容器对齐
justify/align-items: 对齐方式;          // 项目默认对齐

/* 项目核心属性 */
grid-area: 区域名 | 行列起止;           // 区域/位置
grid-column/row: span 数量 | 线编号;    // 行列跨度
justify/align-self: 对齐方式;          // 单个项目对齐

/* 高级函数 */
repeat(次数/auto-fit, 尺寸);            // 重复定义
minmax(最小值, 最大值);                 // 尺寸范围
clamp(最小值, 首选值, 最大值);          // 自适应尺寸
