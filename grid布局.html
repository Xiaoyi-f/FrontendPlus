Grid布局核心语法速查表

/* 容器核心属性 */
display: grid | inline-grid;                // 启用Grid布局
grid-template-columns: 100px 1fr minmax(100px, 1fr); // 列定义
grid-template-rows: repeat(3, 1fr);        // 行定义（重复函数）
grid-template-areas: "a a b" "c d d";      // 区域命名
grid-gap: 10px 15px;                       // 间距（行+列）
grid-auto-rows: 80px;                       // 隐式网格行高
grid-auto-flow: row | column | dense;      // 自动流方向
justify-content: start | center | space-between; // 容器水平对齐
align-content: start | center | space-around;    // 容器垂直对齐
justify-items: start | center | end;       // 项目水平对齐
align-items: start | center | end;         // 项目垂直对齐

/* 项目核心属性 */
grid-area: header;                         // 关联命名区域
grid-column: 1 / 3 | span 2;               // 列跨度（网格线/跨列数）
grid-row: 2 / 4 | span 2;                  // 行跨度（网格线/跨行数）
justify-self: start | center | end;        // 单个项目水平对齐
align-self: start | center | end;          // 单个项目垂直对齐

/* 高级函数 */
repeat(auto-fit, minmax(200px, 1fr));      // 响应式自动适配列数
clamp(100px, 50%, 300px);                  // 自适应尺寸（最小/首选/最大）
fit-content(200px);                        // 自适应不超过最大值
